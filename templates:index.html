<head>

  <meta charset="UTF-8" />

  <meta name="viewport" content="width=device-width, initial-scale=1.0" />

  <title>Produktspezifikation Generator</title>

  <script defer>

    async function handleSubmit(e) {

      e.preventDefault();

      const status = document.getElementById("status");

      const downloadLink = document.getElementById("download-link");

      downloadLink.style.display = "none";

      status.innerText = " Dokument wird generiert...";


      const form = document.getElementById("upload-form");

      const formData = new FormData(form);


      try {

        const response = await fetch("/fill-doc", {

          method: "POST",

          body: formData

        });


        if (!response.ok) throw new Error("Fehler beim Generieren");


        const blob = await response.blob();

        const url = window.URL.createObjectURL(blob);


        downloadLink.href = url;

        downloadLink.download = "filled_specification.docx";

        downloadLink.style.display = "inline-block";

        status.innerText = " Erfolgreich generiert!";

      } catch (err) {

        console.error(err);

        status.innerText = " Fehler: " + err.message;

      }

    }

  </script>

  <style>

    body {

      font-family: sans-serif;

      max-width: 600px;

      margin: 3rem auto;

    }

      margin-bottom: 1rem;

    }

    #download-link {

      margin-top: 1rem;

      display: none;

    }

    <textarea name="disclaimer" rows="4" style="width: 100%;"></textarea><br />


    <button type="submit"> Dokument generieren</button>

  </form>


  <p id="status"></p>
